\[
0 = \dot{m} (h_{\text{ein}} - h_{\text{aus}}) + \dot{Q}_R + \dot{Q}_{\text{aus}}
\]

\[
-\dot{Q}_{\text{aus}} = \dot{m} (h_{\text{ein}} - h_{\text{aus}}) + \dot{Q}_R \quad \text{(1 - $\dot{Q}_{\text{aus}}$)}
\]

\[
\dot{Q}_{\text{aus}} = \dot{m} (h_{\text{aus}} - h_{\text{ein}}) - \dot{Q}_R
\]

\[
