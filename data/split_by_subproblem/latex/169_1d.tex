\section*{Problem d)}

\begin{equation*}
\frac{dE}{dt} = \dot{m}(h_1 - h_2) + \sum \dot{Q} - \sum \dot{W}
\end{equation*}

\begin{equation*}
\Delta E = \Delta m_{12} (h_1 - h_2)
\end{equation*}

\begin{equation*}
\Delta E = \Delta U = U_2 - U_1
\end{equation*}

\begin{equation*}
\Rightarrow \Delta m_{12} = \frac{u_2 - u_1}{h_1 - h_2} = \frac{c(T_2 - T_1)}{c(T_1 - T_2)} = \infty
\end{equation*}
