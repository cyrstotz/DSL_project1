\[
0 = \dot{m}[h_e - h_a] + \dot{Q}_R
\]

\[
\dot{m} = \frac{\dot{Q}_R}{h_a - h_e} = 65 \, \text{kW}
\]

\[
\Sigma Q = \Delta E = U + KE + PE
\]

\[

\]

\[
\Delta E = m c \Delta T
