\[
\dot{Q}_{\text{out}} = \dot{Q}_R - \dot{m}_{\text{in}} (h_{\text{out}} - h_{\text{in}})
\]

\[
\dot{Q}_{\text{out}} = \dot{Q}_R - \dot{m}_{\text{in}} = 100 \, \text{kW} - 42.08 \, \text{kW} = 57.92 \, \text{kW}
\]

\[
