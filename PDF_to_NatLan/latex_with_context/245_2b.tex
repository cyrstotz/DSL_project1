\[
\frac{T_6}{T_5} = \left( \frac{p_6}{p_5} \right)^{\frac{\kappa - 1}{\kappa}}

\[
T_6 = T_5 \cdot \left( \frac{0.191}{0.5} \right)^{\frac{0.4}{1.4}} = 328.07 \, \text{K}
\]
