\[
0 = -\Delta ex_{\text{str}} + \left( 1 - \frac{T_0}{T_B} \right) q_B - ex_{\text{vol}}

\[
ex_{\text{vol}} = \left( 1 - \frac{T_0}{T_B} \right) q_B - \Delta ex_{\text{str}}
