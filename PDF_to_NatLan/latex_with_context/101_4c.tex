\[

\]

\[

\[
h_g(3.3765 \, \text{bar}) = \frac{h_g(3.6 \, \text{bar}) - h_g(3.2 \, \text{bar})}{3.6 - 3.2} \cdot (3.3765 - 3.2) + h_g(3.2 \, \text{bar})

\[
x = \frac{h_1 - h_f}{h_g - h_f} = \frac{93.42 - 55.8}{249.5072 - 55.8} = 0.1963
\]
