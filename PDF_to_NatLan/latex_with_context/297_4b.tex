\[
p_2 = 3 \, \text{bar}
\]
\[
h_2 = h_{3 \, \text{bar}}
\]
\[
h_2 = h_{3 \, \text{bar}} + \frac{3 \, \text{bar} - 2.5 \, \text{bar}}{3.2 \, \text{bar} - 2.5 \, \text{bar}} \cdot (h_{2.5 \, \text{bar}} - h_{3.2 \, \text{bar}})
\]

\[
Q = \dot{m} \cdot (h_2 - h_3) - \dot{W}_K
\]
\[
\dot{W}_K = \frac{\dot{Q}}{\epsilon_K}
\]
