\[
m_2 u_2 - m_1 u_1 = \Delta m_{\text{in}} h_{\text{ein}} + Q_{12} - W_{12}
\]

\[
(m_1 + \Delta m_{\text{in}}) u_2 - m_1 u_1 = \Delta m_{\text{in}} h_{\text{ein}}
\]

\[
h_{\text{ein}} = 83.06 \, \frac{\text{kJ}}{\text{kg}} \quad \text{(from Table A-2)}

\[
