\[
m \left( u_{2g} - u_{1g} \right) + Q_{12} = 0

\[
Q_{12} = m_g c_V \left( T_{g,2} - T_{g,1} \right)

\[
Q_{12} = 0.02777 \, \text{kg} \cdot 0.633 \, \text{kJ/kg·K} \cdot \left( -499.997 \, \text{K} \right) = -7.21 \, \text{kJ}
\]
