\[
\Delta ex_{\text{STR}} = ex_{\text{STR},6} - ex_{\text{STR},0}.

\[
ex_{\text{STR}} = h_6 - h_0 - T_0 (s_6 - s_0) + \frac{w_6^2 - w_0^2}{2},

\[
