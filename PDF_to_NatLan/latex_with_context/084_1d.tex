\[

\[
m_{2} \cdot u_{2} - m_{1} \cdot u_{1} = \dot{Q}_{\text{out}}
\]

\[
\Delta m_{12} \cdot (u_{2} + u_{1}) = \dot{Q}_{\text{out}}

\[
\Delta m_{12} = \frac{\dot{Q}_{\text{out}}}{u_{2} + u_{1}} = \frac{35 \, \text{MJ}}{399.37}
