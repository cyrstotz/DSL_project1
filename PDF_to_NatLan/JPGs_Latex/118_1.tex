\[
\dot{m} = 0.3 \, \text{kg/s}, \quad s_e = 0.2545 \, \frac{\text{kJ}}{\text{kg·K}}, \quad s_a = 1.377 \, \frac{\text{kJ}}{\text{kg·K}}

\[
\dot{Q} = \dot{Q}_R - \dot{Q}_{\text{out}}

\[
