\[
(m_1 + \Delta m_{12}) U_2 - m_1 U_1 = \Delta m_{12} \, h_{\text{ein,12}} - Q_{\text{aus,12}}
\]

\[
\Delta m_{12} (U_2 - h_{\text{ein,12}}) = m_1 U_1 - m_1 U_2 - Q_{\text{aus,12}}
\]  
\[
\Delta m_{12} = \frac{m_1 U_1 - m_1 U_2 - Q_{\text{aus,12}}}{U_2 - h_{\text{ein,12}}}
\]

\[
