\[

\[
p_{g,1} = p_{\text{EW},1} + \frac{m_K \cdot g}{A} + p_{\text{amb}},

\[
pV = mRT.

\[
T_{g,1} = 773.15 \, \text{K}, \quad p_{g,1} = 1.40 \cdot 10^5 \, \text{Pa}, \quad V_{g,1} = 3.14 \cdot 10^{-3} \, \text{m}^3, \quad R = 0.166 \, \text{kJ/kg路K}.

\[
