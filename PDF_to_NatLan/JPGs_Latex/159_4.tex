\[
\Delta E_{\text{EW}} = \Delta U_{\text{EW}} = Q_{12}

\[
u_{\text{EW}} - u_{\text{EW,ref}} + Q_{12} = \frac{1500}{m_{\text{EW}}} = \frac{75 \, \text{kJ}}{\text{kg}}

\[
u_2 = (x_2 \cdot u_{\text{rest}}[0^\circ] + (1 - x_2) \cdot u_{\text{fusion}}[0^\circ]) \, \text{EW}

\[
