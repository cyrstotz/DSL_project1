\[
\frac{dE}{dt} = \dot{Q} + \dot{W} - \dot{m} \cdot \Delta h

\[
m \cdot \frac{du}{dt} = Q
\]

\[
u_2 = \frac{Q}{m \cdot u_{\text{ln}}}

\[
