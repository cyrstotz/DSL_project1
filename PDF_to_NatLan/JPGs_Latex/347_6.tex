\[
\epsilon_K = \frac{\lvert \dot{Q}_K \rvert}{\lvert \dot{W}_K \rvert} = \frac{\lvert \dot{Q}_K \rvert}{W_K}
\]

\[
0 = \dot{m} (h_1 - h_2) + \dot{Q}_K - W_K
\]

\[
\epsilon_K = \frac{\dot{m} (h_2 - h_1)}{W_K}
