\[
U_{1,\text{EW}} = U_{\text{flüssig}} + x_{\text{Eis}} \cdot (U_{\text{fest}} - U_{\text{flüssig}})

\[

\[
U_{2,\text{EW}} = \frac{Q_k}{m_w} + U_{1,\text{EW}}

\[
-191.6 = -0.033 + x_{\text{Eis,2}} \cdot (-333.442 + 0.033)
\]

\[
x_{\text{Eis,2}} = 0.575
